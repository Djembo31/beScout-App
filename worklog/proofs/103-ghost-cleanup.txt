WITH ghosts AS (
  SELECT p.id FROM players p
  LEFT JOIN clubs c ON c.id = p.club_id
  LEFT JOIN leagues l ON l.id = c.league_id
  WHERE (p.nationality IS NULL OR p.nationality = '')
    AND (l.short != 'TFF1' OR l.short IS NULL)
    AND p.matches = 0
    AND (p.last_appearance_gw IS NULL OR p.last_appearance_gw = 0)
)
UPDATE players SET club_id = NULL, updated_at = NOW()
WHERE id IN (SELECT id FROM ghosts) RETURNING id;
