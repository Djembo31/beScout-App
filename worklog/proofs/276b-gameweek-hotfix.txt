VERIFY-COMMAND post-Deploy:
  -- nach jedem täglich-06:00-Cron-Run:
  SELECT l.name, l.active_gameweek,
         MAX(f.gameweek) FILTER (WHERE f.status IN ('finished','simulated')) AS last_finished,
         MIN(f.gameweek) FILTER (WHERE f.status NOT IN ('finished','simulated')) AS first_open
  FROM leagues l
  LEFT JOIN fixtures f ON f.league_id = l.id
  GROUP BY l.id, l.name, l.active_gameweek
  ORDER BY l.name;
  -- Erwartet: active_gw = first_open (außer Postponed-Edge-Cases).
